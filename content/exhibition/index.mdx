import glossaryData from "../data/glossaryData";
import { renderWithGlossary } from "../utils/renderWithGlossary";

export default function HistoryPage() {
  const content = `
    NASA worked closely with the EPA and NOAA during this period.
    The TVA played a key role in regional development.
  `;

  return (
    <main>
      <article>
        <p>{renderWithGlossary(content, glossaryData)}</p>
      </article>

      {/* Glossary section */}
      <section className="glossary-section">
        <h2>Glossary</h2>
        {Object.entries(glossaryData).map(([term, definition]) => (
          <div key={term} id={`glossary-${term}`} className="glossary-item">
            <strong>{term}</strong>: {definition}
          </div>
        ))}
      </section>
    </main>
  );
}
